Mobilepopup - jQuery pop-up plugin




Example (demo):


$("body").on("click",".popup-form",function(){
    $.mobilepopup({
		targetblock:".pop-up-form",
		width:"300px",
		height:"300px",
		onformsubmited: function(data, el){
			$.mobilepopup('reload',{html:data});
		}
	});
    return false;
});
